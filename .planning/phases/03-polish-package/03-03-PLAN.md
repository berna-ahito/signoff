From frontend/src/hooks/useAuth.ts (full source):
```typescript
export function useAuth() {
  const [token, setToken] = useState<string | null>(() => localStorage.getItem('access_token'))
  const [role, setRole] = useState<Role | null>(...)

  const login = useCallback(async (email: string, password: string) => {
    const data = await apiLogin(email, password)
    const parsedRole = parseRole(data.access_token)
    localStorage.setItem('access_token', data.access_token)
    setToken(data.access_token); setRole(parsedRole)
    return parsedRole
  }, [])

  const logout = useCallback(() => {
    localStorage.removeItem('access_token')
    setToken(null); setRole(null)
  }, [])

  return { token, role, isAuthenticated: !!token, login, logout }
}
```
